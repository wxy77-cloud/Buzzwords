        function getYearData(year) {
            return timelineData.filter(item => item.value[0] === year);

        function getYearlyAggregatedData() {
            const yearlyData = {};
            timelineData.forEach(item => {
                const year = item.value[0];
                if (!yearlyData[year]) {
                    yearlyData[year] = { count: 0, totalHot: 0, items: [] };
                }
                yearlyData[year].count++;
                yearlyData[year].totalHot += item.value[2];
                yearlyData[year].items.push(item);
            });
            
            const result = [];
            for (let year = 2019; year <= 2026; year++) {
                if (yearlyData[year]) {
                    const avgHot = Math.round(yearlyData[year].totalHot / yearlyData[year].count);
                    result.push({
                        year: year,
                        avgHot: avgHot,
                        count: yearlyData[year].count,
                        items: yearlyData[year].items
                    });
                } else {
                    result.push({ year: year, avgHot: 0, count: 0, items: [] });
                }
            }
            return result;
        }

        function initTimelineSlider() {
            const slider = document.getElementById('timelineSlider');

                label.textContent = year;
                label.style.left = `${(index / (years.length - 1)) * 100}%`;
                label.addEventListener('click', () => {
                        // 清除之前的选中状�?                        document.querySelectorAll('.timeline-label').forEach(l => l.classList.remove('active'));
                        label.classList.add('active');
                        
                        // 更新视图为月份视�?                        currentView = 'month';
                        updateMonthView(parseInt(year));
                        
                        // 显示返回按钮
                        const backBtn = document.querySelector('.timeline-back-btn');
                        if (backBtn) {
                            backBtn.style.display = 'block';
                        }
                    });
                labels.appendChild(label);
            });

            // 添加返回按钮
            const backBtn = document.createElement('div');
            backBtn.className = 'timeline-back-btn';
            backBtn.textContent = '�?返回年度视图';
            backBtn.style.display = 'none';
            backBtn.addEventListener('click', () => {
                document.querySelectorAll('.timeline-label').forEach(l => l.classList.remove('active'));
                backBtn.style.display = 'none';
                currentView = 'year';
                selectedYear = null;
                updateYearView();
            });
            slider.appendChild(backBtn);

            // 暴露backBtn供外部使�?            window.timelineBackBtn = backBtn;
        }
